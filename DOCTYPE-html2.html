// parse numbers string (e.g. "5,10,22")
function parseNumbers(str){
if(!str) return [];
return Array.from(new
Set(str.split(",").map(s=>s.trim()).filter(Boolean).map(Number).filter(n=>Number.
isInteger(n) && n>=1 && n<=500)));
}
// BUY numbers (1 ou mais)
async function buyNumbers(){
const user = auth.currentUser;
if(!user){ setText("buy-msg","Faa login primeiro."); return; }
// selectedSet + manual input
const manual = document.getElementById("manual-numbers").value.trim();
const combined = new Set();
// add selected numbers from grid
document.querySelectorAll(".num-btn.selected").forEach(b=>
combined.add(Number(b.dataset.num)));
// add manual
if(manual){
parseNumbers(manual).forEach(n=> combined.add(n));
}
const numbers = Array.from(combined).sort((a,b)=>a-b);
if(numbers.length === 0){ setText("buy-msg","Escolha pelo menos 1 nmero
(1500)."); return; }
try{
// check conflicts: scan compras for pendente/aprovado to find overlapping
numbers
const allSnap = await getDocs(collection(db,"compras"));
const conflicts = [];
allSnap.forEach(d=>{
const p = d.data();
if(p.status==="pendente" || p.status==="aprovado"){
const nums = Array.isArray(p.number)?p.number:[p.number];
nums.forEach(n => { if(numbers.includes(Number(n))) conflicts.push(n);
});
}
});
if(conflicts.length){
setText("buy-msg"," Os nmeros j reservados/aprovados: " + [...new
Set(conflicts)].join(", "));
return;
}
// create purchase
const docRef = await addDoc(collection(db,"compras"), {
userId: user.uid,
number: numbers,
price: 750 * numbers.length,
status: "pendente",
ts: serverTimestamp(),
pdfUrl: null,
resposta: ""
});
// upload PDF if provided
const file = document.getElementById("file-input").files[0];
if(file){
const sref = storageRef(storage,
`comprovativos/${user.uid}/${docRef.id}.pdf`);
await uploadBytes(sref, file, { contentType: "application/pdf" });
const url = await getDownloadURL(sref);
await updateDoc(doc(db,"compras",docRef.id), { pdfUrl: url });
}
setText("buy-msg"," Compra registada! Aguarde validao do
administrador.");
document.getElementById("manual-numbers").value = "";
document.getElementById("file-input").value = "";
// clear selections
document.querySelectorAll(".num-btn.selected").forEach(b=>
b.classList.remove("selected"));
await renderMyPurchases(user.uid);
await renderAdmin(); // atualiza admin
}catch(e){
setText("buy-msg","Erro: " + (e.message||e));
}
}

}
